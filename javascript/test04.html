        function solution(a, b, c){
            let answer = "YES", max;
            let total = a + b + c;
    
            if(a > b) max = a;
            else max = b;
            if(c > max) max = c;
            if(total-max <= max) answer = "NO"; 
            
            return answer;
        }

    }

20. 다음을 결과값을 작성하시오.

{
        function solution(day, arr){
            let answer = 0;
            for(let x of arr){
                if(x % 10 == day) answer++;
            }
            return answer;
        }
        
        arr = [25, 23, 11, 47, 53, 17, 33, 40];
        console.log(solution(0, arr));
    }
